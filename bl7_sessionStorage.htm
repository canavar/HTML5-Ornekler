Session Storage Lab
function setToSessionStorage(text) {
   // sessionStorage.setItem('demoVariable', text);
   sessionStorage.demoVariable = text;
}

function getFromSessionStorage() {
   // return sessionStorage.getItem('demoVariable');
   return sessionStorage.demoVariable;
}

function deleteFromSessionStorage() {
   // sessionStorage.removeItem('demoVariable');
   delete sessionStorage.demoVariable;
}

function clearSessionStorage() {
   var storageLength = sessionStorage.length;
   sessionStorage.clear();
   alert(storageLength + ' tane kayıt sessionStorage\'dan silindi.');
}

function listSessionStorage() {
   for (var i = 0; i < sessionStorage.length; i++) {
      var currentKey = sessionStorage.key(i);
      alert(currentKey + ' : ' + sessionStorage.getItem(currentKey));
   }
}
